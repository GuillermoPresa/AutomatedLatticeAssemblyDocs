42class AsyncToken
43{
44 public:
45 AsyncToken() : set(NULL) {}
46 AsyncToken(const AsyncToken& copy) : set(copy.set) {}
47 AsyncToken(const MaskSet& _set) : set(&_set) {}
48
49 inline AsyncToken& operator=(const AsyncToken& t) { set = t.set; return *this; }
50 inline AsyncToken& operator=(const MaskSet& s) { set = &s; return *this; }
51
52 // clear the async token
53 inline void clear() { set = NULL; }
54
55 inline bool isValid() const { return set!=NULL; }
56 inline bool isActive() const { return set && set->isActive(); }
57 inline bool isComplete() const { return set && set->isComplete(); }
58 inline bool isUnresponsive() const { return set && set->isUnresponsive(); }
59
60 // returns an async token that indicates an unresponsive device/servo
61 static AsyncToken Unresponsive() {
62 static MaskSet um(0, LssQuery);
63 return AsyncToken(um);
64 }
65
66protected:
67 const MaskSet* set;
68};
