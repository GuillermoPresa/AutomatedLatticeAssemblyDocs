257 private void HeightCalculation()
258 {
259 float distanceToGround = DistanceToGround();
260 if(useScrollwheelZooming)
261 zoomPos += ScrollWheel * Time.deltaTime * scrollWheelZoomingSensitivity;
262 if (useKeyboardZooming)
263 zoomPos += ZoomDirection * Time.deltaTime * keyboardZoomingSensitivity;
264
265 zoomPos = Mathf.Clamp01(zoomPos);
266
267 float targetHeight = Mathf.Lerp(minHeight, maxHeight, zoomPos);
268 float difference = 0;
269
270 if(distanceToGround != targetHeight)
271 difference = targetHeight - distanceToGround;
272
273 m_Transform.position = Vector3.Lerp(m_Transform.position,
274 new Vector3(m_Transform.position.x, targetHeight + difference, m_Transform.position.z), Time.deltaTime * heightDampening);
275 }
