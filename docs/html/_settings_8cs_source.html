478 public static bool GetLock(DoF doF, Axis axis) {
479 Locks translationLocks = Mode == OperationMode.Fly || Mode == OperationMode.Orbit ? NavTranslationLock : ManipulateTranslationLock;
480 Locks rotationLocks = Mode == OperationMode.Fly || Mode == OperationMode.Orbit ? NavRotationLock : ManipulateRotationLock;
481 Locks locks = doF == DoF.Translation ? translationLocks : rotationLocks;
482
483 switch (axis) {
484 case Axis.X:
485 return (locks.X || locks.All) && !Application.isPlaying;
486 case Axis.Y:
487 return (locks.Y || locks.All) && !Application.isPlaying;
488 case Axis.Z:
489 return (locks.Z || locks.All) && !Application.isPlaying;
490 default:
491 throw new ArgumentOutOfRangeException("axis");
492 }
493 }
